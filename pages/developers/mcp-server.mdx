# MCP Server Development Guide

This comprehensive guide covers the development, architecture, and implementation details of Lokus's Model Context Protocol (MCP) server, which provides 68 sophisticated tools for AI assistant integration.

## Architecture Overview

### MCP Protocol Implementation

The Lokus MCP server implements the industry-standard Model Context Protocol specification, enabling seamless integration with AI assistants like Claude, GPT-4, and other MCP-compatible clients.

```
AI Assistant ←→ MCP Client ←→ stdio ←→ MCP Server ←→ Lokus Core
```

## Core Server Implementation

### Main Server (`server.js`)

```javascript
#!/usr/bin/env node

import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';
import { 
  ListToolsRequestSchema,
  CallToolRequestSchema,
  ErrorCode,
  McpError 
} from '@modelcontextprotocol/sdk/types.js';

// Import all tool categories
import { NoteTools } from './tools/noteTools.js';
import { WorkspaceTools } from './tools/workspaceTools.js';
import { SearchTools } from './tools/searchTools.js';
import { AITools } from './tools/aiTools.js';
import { FileTools } from './tools/fileTools.js';
import { EditorTools } from './tools/editorTools.js';
import { SystemTools } from './tools/systemTools.js';

class LokusMCPServer {
  constructor() {
    this.server = new Server(
      {
        name: 'lokus-mcp-server',
        version: '1.0.0',
      },
      {
        capabilities: {
          tools: {},
        },
      }
    );

    this.toolCategories = [
      new NoteTools(),
      new WorkspaceTools(),
      new SearchTools(),
      new AITools(),
      new FileTools(),
      new EditorTools(),
      new SystemTools()
    ];

    this.setupHandlers();
    this.registerTools();
  }

  setupHandlers() {
    // List available tools
    this.server.setRequestHandler(ListToolsRequestSchema, async () => {
      const allTools = [];
      
      for (const category of this.toolCategories) {
        const tools = await category.getTools();
        allTools.push(...tools);
      }

      return { tools: allTools };
    });

    // Execute tool calls
    this.server.setRequestHandler(CallToolRequestSchema, async (request) => {
      const { name, arguments: args } = request.params;

      try {
        // Find the category that handles this tool
        for (const category of this.toolCategories) {
          const tools = await category.getTools();
          const tool = tools.find(t => t.name === name);
          
          if (tool) {
            const result = await category.executeTool(name, args || {});
            return {
              content: [
                {
                  type: 'text',
                  text: typeof result === 'string' ? result : JSON.stringify(result, null, 2)
                }
              ]
            };
          }
        }

        throw new McpError(
          ErrorCode.MethodNotFound,
          `Tool not found: ${name}`
        );
      } catch (error) {
        if (error instanceof McpError) {
          throw error;
        }

        throw new McpError(
          ErrorCode.InternalError,
          `Tool execution failed: ${error.message}`
        );
      }
    });
  }

  async registerTools() {
    // Initialize all tool categories
    for (const category of this.toolCategories) {
      await category.initialize();
    }
  }

  async start() {
    const transport = new StdioServerTransport();
    await this.server.connect(transport);
    
    console.error('Lokus MCP Server started successfully');
    console.error(`Loaded ${this.toolCategories.length} tool categories`);
    
    // Count total tools
    let totalTools = 0;
    for (const category of this.toolCategories) {
      const tools = await category.getTools();
      totalTools += tools.length;
    }
    console.error(`Total tools available: ${totalTools}`);
  }
}

// Start the server
const server = new LokusMCPServer();
server.start().catch(error => {
  console.error('Failed to start MCP server:', error);
  process.exit(1);
});
```

```javascript
    this.categoryName = categoryName;

  }

    // Must be implemented by subclasses

    // Must be implemented by subclasses
    throw new Error('executeTool() must be implemented');
  }

  validateArgs(args, schema) {
    // Basic validation - can be enhanced with JSON schema
    for (const [key, requirements] of Object.entries(schema)) {
      if (requirements.required && !(key in args)) {
        throw new Error(`Missing required parameter: ${key}`);
      }
      
      if (key in args && requirements.type) {
        const actualType = typeof args[key];
        if (actualType !== requirements.type) {
          throw new Error(`Parameter ${key} must be of type ${requirements.type}, got ${actualType}`);
        }
      }
    }
  }

  createToolSchema(name, description, properties, required = []) {
    return {
      name,
      description,
      inputSchema: {
        type: 'object',
        properties,
        required
      }
    };
  }
}
```

```javascript
      tags,

    const finalContent = content || templateContent;
    const fullContent = `${frontmatterYaml}\n\n${finalContent}`;

    // Ensure directory exists
    await fs.mkdir(path.dirname(fullPath), { recursive: true });

    // Write file
    await fs.writeFile(fullPath, fullContent, 'utf8');

    return {
      success: true,
      path: fileName,
      fullPath,
      title,
      created: noteMetadata.created
    };
  }

  async updateNote(args) {
    this.validateArgs(args, {
      path: { required: true, type: 'string' }
    });

    const { 
      path: notePath, 
      content, 
      appendContent, 
      prependContent,
      addTags = [],
      removeTags = [],
      updateFrontmatter = {},
      createBackup = true
    } = args;

    const fullPath = path.join(this.workspace.path, notePath);

    // Check if file exists
    try {
      await fs.access(fullPath);
    } catch {
      throw new Error(`Note not found: ${notePath}`);
    }

    // Create backup if requested
    if (createBackup) {
      const backupPath = `${fullPath}.backup.${Date.now()}`;
      await fs.copyFile(fullPath, backupPath);
    }

    // Read current content
    const currentContent = await fs.readFile(fullPath, 'utf8');
    const { frontmatter, body } = this.parseFrontmatter(currentContent);

    // Update frontmatter
    let newFrontmatter = { ...frontmatter, ...updateFrontmatter };
    
    // Handle tags
    let currentTags = newFrontmatter.tags || [];
    currentTags = [...currentTags, ...addTags];
    currentTags = currentTags.filter(tag => !removeTags.includes(tag));
    newFrontmatter.tags = [...new Set(currentTags)]; // Remove duplicates

    // Update modified time
    newFrontmatter.modified = new Date().toISOString();

    // Update content
    let newBody = body;
    if (content !== undefined) {
      newBody = content;
    }
    if (prependContent) {
      newBody = prependContent + '\n\n' + newBody;
    }
    if (appendContent) {
      newBody = newBody + '\n\n' + appendContent;
    }

    // Generate final content
    const frontmatterYaml = this.generateFrontmatter(newFrontmatter);
    const finalContent = `${frontmatterYaml}\n\n${newBody}`;

    // Write updated content
    await fs.writeFile(fullPath, finalContent, 'utf8');

    return {
      success: true,
      path: notePath,
      modified: newFrontmatter.modified,
      changes: {
        contentUpdated: content !== undefined || appendContent || prependContent,
        tagsAdded: addTags.length > 0,
        tagsRemoved: removeTags.length > 0,
        frontmatterUpdated: Object.keys(updateFrontmatter).length > 0
      }
    };
  }

  // Helper methods
  validateWorkspace() {
    if (!this.workspace) {
      throw new Error('No active workspace found');
    }
  }

  sanitizeFileName(title) {
    return title
      .toLowerCase()
      .replace(/[^a-z0-9\s-]/g, '')
      .replace(/\s+/g, '-')
      .substring(0, 50);
  }

  async loadTemplate(templateName) {
    const templateMap = {
      basic: '# {{title}}\n\n',
      daily: '# Daily Note - {{date}}\n\n## Tasks\n\n## Notes\n\n## Reflections\n\n',
      meeting: '# {{title}}\n\n**Date:** {{date}}\n**Attendees:** \n\n## Agenda\n\n## Discussion\n\n## Action Items\n\n',
      project: '# {{title}}\n\n## Overview\n\n## Goals\n\n## Timeline\n\n## Resources\n\n## Progress\n\n',
      research: '# {{title}}\n\n## Research Question\n\n## Methodology\n\n## Findings\n\n## Sources\n\n## Conclusions\n\n',
      review: '# {{title}} - Review\n\n## Summary\n\n## What Worked\n\n## What Didn\'t Work\n\n## Lessons Learned\n\n## Next Steps\n\n',
      empty: ''
    };

    return templateMap[templateName] || templateMap.basic;
  }

  generateFrontmatter(metadata) {
    const yaml = Object.entries(metadata)
      .map(([key, value]) => {
        if (Array.isArray(value)) {
          return `${key}:\n${value.map(v => `  - ${v}`).join('\n')}`;
        }
        return `${key}: ${JSON.stringify(value)}`;
      })
      .join('\n');

    return `---\n${yaml}\n---`;
  }

  parseFrontmatter(content) {
    const frontmatterRegex = /^---\n([\s\S]*?)\n---\n([\s\S]*)$/;
    const match = content.match(frontmatterRegex);

    if (!match) {
      return { frontmatter: {}, body: content };
    }

    const frontmatterText = match[1];
    const body = match[2];

    // Simple YAML parsing (in production, use a proper YAML parser)
    const frontmatter = {};
    frontmatterText.split('\n').forEach(line => {
      const [key, ...valueParts] = line.split(':');
      if (key && valueParts.length > 0) {
        const value = valueParts.join(':').trim();
        try {
          frontmatter[key.trim()] = JSON.parse(value);
        } catch {
          frontmatter[key.trim()] = value.replace(/^["']|["']$/g, '');
        }
      }
    });

    return { frontmatter, body };
  }

  // Additional methods for other note operations...
}
```

```javascript
  async searchContent(args) {
    const {
      query,
      fileTypes = ['.md', '.txt'],
      caseSensitive = false,
      useRegex = false,
      wholeWord = false,
      maxResults = 50,
      contextLines = 3,
      excludePaths = []
    } = args;

    const results = [];
    const searchRegex = this.buildSearchRegex(query, { caseSensitive, useRegex, wholeWord });
    
    // Get all files in workspace
    const files = await this.getAllFiles(this.workspace.path);
    
    for (const filePath of files) {
      // Skip excluded paths
      if (excludePaths.some(exclude => filePath.includes(exclude))) {
        continue;
      }

      // Filter by file type
      if (fileTypes.length > 0 && !fileTypes.some(ext => filePath.endsWith(ext))) {
        continue;
      }

      try {
        const content = await fs.readFile(filePath, 'utf8');
        const lines = content.split('\n');
        
        for (let i = 0; i < lines.length; i++) {
          const line = lines[i];
          const match = line.match(searchRegex);
          
          if (match) {
            const startLine = Math.max(0, i - contextLines);
            const endLine = Math.min(lines.length - 1, i + contextLines);
            const context = lines.slice(startLine, endLine + 1);
            
            results.push({
              file: path.relative(this.workspace.path, filePath),
              line: i + 1,
              match: match[0],
              context: context.join('\n'),
              excerpt: line.trim()
            });

            if (results.length >= maxResults) {
              break;
            }
          }
        }

        if (results.length >= maxResults) {
          break;
        }
      } catch (error) {
        // Skip files that can't be read
        continue;
      }
    }

    return {
      query,
      totalResults: results.length,
      results,
      searchOptions: { caseSensitive, useRegex, wholeWord, fileTypes }
    };
  }

  async getGraphData(args) {
    const {
      maxNodes = 500,
      filterByType = ['document', 'placeholder'],
      includeMetrics = true
    } = args;

    const nodes = new Map();
    const edges = [];
    
    // Scan all markdown files
    const files = await this.getAllFiles(this.workspace.path, ['.md']);
    
    for (const filePath of files.slice(0, maxNodes)) {
      const relativePath = path.relative(this.workspace.path, filePath);
      const fileName = path.basename(filePath, '.md');
      
      // Add document node
      nodes.set(relativePath, {
        id: relativePath,
        label: fileName,
        type: 'document',
        path: relativePath
      });

      try {
        const content = await fs.readFile(filePath, 'utf8');
        
        // Extract wiki links
        const wikiLinkRegex = /\[\[([^\]]+)\]\]/g;
        let match;
        
        while ((match = wikiLinkRegex.exec(content)) !== null) {
          const linkTarget = match[1];
          const targetPath = this.resolveWikiLink(linkTarget);
          
          // Add target node if not exists
          if (!nodes.has(targetPath)) {
            nodes.set(targetPath, {
              id: targetPath,
              label: linkTarget,
              type: 'placeholder',
              path: targetPath
            });
          }
          
          // Add edge
          edges.push({
            source: relativePath,
            target: targetPath,
            type: 'wikilink'
          });
        }
      } catch (error) {
        // Skip files that can't be read
        continue;
      }
    }

    // Filter nodes by type
    const filteredNodes = Array.from(nodes.values())
      .filter(node => filterByType.includes(node.type));

    // Calculate metrics if requested
    if (includeMetrics) {
      for (const node of filteredNodes) {
        const inDegree = edges.filter(e => e.target === node.id).length;
        const outDegree = edges.filter(e => e.source === node.id).length;
        
        node.metrics = {
          inDegree,
          outDegree,
          totalDegree: inDegree + outDegree
        };
      }
    }

    return {
      nodes: filteredNodes,
      edges,
      metrics: {
        totalNodes: filteredNodes.length,
        totalEdges: edges.length,
        nodeTypes: this.countByProperty(filteredNodes, 'type')
      }
    };
  }

  // Helper methods
  buildSearchRegex(query, options) {
    let pattern = query;
    
    if (!options.useRegex) {
      // Escape special regex characters
      pattern = query.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    }
    
    if (options.wholeWord) {
      pattern = `\\b${pattern}\\b`;
    }
    
    const flags = options.caseSensitive ? 'g' : 'gi';
    return new RegExp(pattern, flags);
  }

  async getAllFiles(dirPath, extensions = null) {
    const files = [];
    
    async function scan(currentPath) {
      const entries = await fs.readdir(currentPath, { withFileTypes: true });
      
      for (const entry of entries) {
        const fullPath = path.join(currentPath, entry.name);
        
        if (entry.isDirectory()) {
          await scan(fullPath);
        } else if (entry.isFile()) {
          if (!extensions || extensions.some(ext => entry.name.endsWith(ext))) {
            files.push(fullPath);
          }
        }
      }
    }
    
    await scan(dirPath);
    return files;
  }

  resolveWikiLink(linkText) {
    // Simple resolution - in practice, this would be more sophisticated
    return linkText.toLowerCase().replace(/\s+/g, '-') + '.md';
  }

  countByProperty(array, property) {
    return array.reduce((counts, item) => {
      const value = item[property];
      counts[value] = (counts[value] || 0) + 1;
      return counts;
    }, {});
  }

  validateWorkspace() {
    if (!this.workspace) {
      throw new Error('No active workspace found');
    }
  }
}
```

## Performance Optimization

### Caching Strategy

```javascript
// utils/cache.js
export class ToolCache {
  constructor(maxSize = 1000, ttl = 300000) { // 5 minutes
    this.cache = new Map();
    this.maxSize = maxSize;
    this.ttl = ttl;
  }

  get(key) {
    const item = this.cache.get(key);
    
    if (!item) return null;
    
    if (Date.now() - item.timestamp > this.ttl) {
      this.cache.delete(key);
      return null;
    }
    
    return item.value;
  }

  set(key, value) {
    if (this.cache.size >= this.maxSize) {
      const firstKey = this.cache.keys().next().value;
      this.cache.delete(firstKey);
    }
    
    this.cache.set(key, {
      value,
      timestamp: Date.now()
    });
  }

  clear() {
    this.cache.clear();
  }
}
```

```javascript
// utils/rateLimit.js
export class RateLimiter {
  constructor(maxRequests = 100, windowMs = 60000) {
    this.requests = new Map();
    this.maxRequests = maxRequests;
    this.windowMs = windowMs;
  }

  isAllowed(clientId) {
    const now = Date.now();
    const windowStart = now - this.windowMs;
    
    if (!this.requests.has(clientId)) {
      this.requests.set(clientId, []);
    }
    
    const clientRequests = this.requests.get(clientId);
    
    // Remove old requests
    while (clientRequests.length > 0 && clientRequests[0] < windowStart) {
      clientRequests.shift();
    }
    
    if (clientRequests.length >= this.maxRequests) {
      return false;
    }
    
    clientRequests.push(now);
    return true;
  }
}
```
