# Plugin Development Guide

Learn how to create powerful plugins for Lokus using our comprehensive plugin architecture with modular APIs, UI panels, data providers, and advanced development tooling.

## Plugin Architecture Overview

Lokus features a professional-grade plugin system with:
- **Modular API Design**: Separated Data, Editor, and UI APIs
- **UI Panel System**: Flexible sidebar, bottom, and modal panels
- **Data Provider Integration**: External service connections
- **React Integration**: Full React component support with hot-reload
- **TypeScript Support**: Complete type safety and IntelliSense
- **CLI Tooling**: Enhanced development workflow with templates
- **Testing Framework**: Plugin compatibility and validation
- **Error Handling**: Robust debugging and recovery systems

## Getting Started

### Prerequisites
- Node.js 18+
- Basic React knowledge
- TypeScript familiarity (recommended)

## Plugin Examples

### 1. UI Panel Plugin with New Architecture

```typescript
// src/index.ts - Main plugin entry
import React from 'react';
import { LokusPlugin, LokusPluginAPI } from '@lokus/plugin-api';
import { TaskManagementPanel } from './panels/TaskManagementPanel';
import { TaskDataProvider } from './api/TaskDataProvider';

const TaskManagementPlugin: LokusPlugin = {
  id: 'task-management',
  name: 'Task Management',
  version: '1.0.0',
  description: 'Advanced task management with kanban boards',
  
  async activate(context: LokusPluginAPI) {
    // Register data provider
    const dataProvider = new TaskDataProvider();
    context.data.registerProvider('tasks', dataProvider);

    // Register UI panel
    context.ui.registerPanel('task-management', {
      title: 'Tasks',
      icon: 'CheckSquare',
      position: 'sidebar-right',
      component: TaskManagementPanel,
      defaultWidth: 300,
      resizable: true
    });

    // Register editor commands
    context.editor.addCommand('insert-task', (attrs) => {
      return context.editor.insertContent(`!task ${attrs.title}`);
    });

    // Subscribe to editor changes for task detection
    context.editor.onContentChange((content) => {
      const tasks = dataProvider.extractTasks(content);
      context.events.emit('tasks-updated', { tasks });
    });
  },

  async deactivate(context: LokusPluginAPI) {
    context.data.unregisterProvider('tasks');
    context.ui.unregisterPanel('task-management');
    context.editor.removeCommand('insert-task');
  }
};

export default TaskManagementPlugin;
```

```typescript
// src/panels/TaskManagementPanel.tsx - UI Panel Component
import React, { useState, useEffect } from 'react';
import { useLokusAPI } from '@lokus/plugin-api';

interface Task {
  id: string;
  title: string;
  status: 'todo' | 'in-progress' | 'done';
  created: Date;
}

export const TaskManagementPanel: React.FC = () => {
  const [tasks, setTasks] = useState<Task[]>([]);
  const lokus = useLokusAPI();

  useEffect(() => {
    const unsubscribe = lokus.events.on('tasks-updated', (data) => {
      setTasks(data.tasks);
    });

    return unsubscribe;
  }, [lokus]);

  const moveTask = (taskId: string, newStatus: Task['status']) => {
    setTasks(prev => prev.map(task => 
      task.id === taskId ? { ...task, status: newStatus } : task
    ));
  };

  const addTask = (title: string) => {
    const newTask: Task = {
      id: Date.now().toString(),
      title,
      status: 'todo',
      created: new Date()
    };
    setTasks(prev => [...prev, newTask]);
  };

  return (
    <div className="task-management-panel">
      <div className="panel-header">
        <h3>Task Management</h3>
        <button onClick={() => addTask('New Task')}>
          Add Task
        </button>
      </div>
      
      <div className="kanban-board">
        {(['todo', 'in-progress', 'done'] as const).map(status => (
          <div key={status} className="kanban-column">
            <h4>{status.replace('-', ' ').toUpperCase()}</h4>
            {tasks
              .filter(task => task.status === status)
              .map(task => (
                <div 
                  key={task.id} 
                  className="task-card"
                  onClick={() => lokus.editor.insertAtCursor(`[[${task.title}]]`)}
                >
                  {task.title}
                </div>
              ))}
          </div>
        ))}
      </div>
    </div>
  );
};
```

import { DataProvider } from '@lokus/plugin-api';

export class TaskDataProvider implements DataProvider {
  id = 'tasks';
  name = 'Task Data Provider';

  async search(query: string): Promise<any[]> {
    // Search through tasks
    const tasks = await this.getAllTasks();
    return tasks.filter(task => 
      task.title.toLowerCase().includes(query.toLowerCase())
    );
  }

  async getAllTasks(): Promise<any[]> {
    // Fetch tasks from storage or external API
    return [];
  }

  extractTasks(content: string): any[] {
    // Extract !task markers from content
    const taskPattern = /!task\s+(.+?)(?:\n|$)/g;
    const tasks = [];
    let match;

    while ((match = taskPattern.exec(content)) !== null) {
      tasks.push({
        id: Date.now().toString() + Math.random(),
        title: match[1].trim(),
        status: 'todo',
        created: new Date()
      });
    }

    return tasks;
  }
}

### 2. Editor API - TipTap Extensions

Extend the editor with custom functionality:

```typescript
// src/api/DiagramExtension.ts
import { Node } from '@tiptap/core';
import { EditorExtension } from '@lokus/plugin-api';

export class DiagramExtension implements EditorExtension {
  createExtension() {
    return Node.create({
      name: 'diagram',
      group: 'block',
      atom: true,

      addAttributes() {
        return {
          type: { default: 'flowchart' },
          code: { default: '' }
        };
      },

      parseHTML() {
        return [{ tag: 'div[data-type="diagram"]' }];
      },

      renderHTML({ HTMLAttributes }) {
        return ['div', { 'data-type': 'diagram', ...HTMLAttributes }];
      },

      addNodeView() {
        return ({ node, editor }) => {
          const container = document.createElement('div');
          container.className = 'diagram-container';
          
          // Render Mermaid diagram
          this.renderDiagram(container, node.attrs.code);
          
          return { dom: container };
        };
      },

      addCommands() {
        return {
          insertDiagram: (attrs) => ({ commands }) => {
            return commands.insertContent({
              type: this.name,
              attrs
            });
          }
        };
      }
    });
  }

  renderDiagram(container: HTMLElement, code: string) {
    // Use Mermaid or other diagram library
    import('mermaid').then(mermaid => {
      mermaid.default.render('diagram', code, (svg) => {
        container.innerHTML = svg;
      });
    });
  }
}
```

### 3. UI API - Panel Management

Create flexible UI panels for custom interfaces:

```typescript
// src/panels/FileExplorerPanel.tsx
import React, { useState, useEffect } from 'react';
import { UIPanel, useLokusAPI } from '@lokus/plugin-api';

interface FileExplorerPanelProps {
  onClose?: () => void;
}

export const FileExplorerPanel: React.FC<FileExplorerPanelProps> = ({ onClose }) => {
  const [files, setFiles] = useState<FileEntry[]>([]);
  const [currentPath, setCurrentPath] = useState('/');
  const lokus = useLokusAPI();

  useEffect(() => {
    loadDirectory(currentPath);
  }, [currentPath]);

  const loadDirectory = async (path: string) => {
    try {
      const entries = await lokus.fs.readDir(path);
      setFiles(entries);
    } catch (error) {
      lokus.notifications.error(`Failed to load directory: ${error.message}`);
    }
  };

  const handleFileClick = async (file: FileEntry) => {
    if (file.is_directory) {
      setCurrentPath(file.path);
    } else {
      // Open file in editor
      const content = await lokus.fs.readFile(file.path);
      lokus.editor.setContent(content);
      lokus.workspace.setActiveFile(file.path);
    }
  };

  const handleCreateFile = async () => {
    const fileName = await lokus.ui.showInputBox({
      placeholder: 'Enter file name...',
      title: 'Create New File'
    });

    if (fileName) {
      const filePath = `${currentPath}/${fileName}`;
      await lokus.fs.writeFile(filePath, '');
      await loadDirectory(currentPath);
    }
  };

  return (
    <div className="file-explorer-panel">
      <div className="panel-header">
        <h3>File Explorer</h3>
        <div className="panel-actions">
          <button onClick={handleCreateFile} title="New File">
            📄
          </button>
          <button onClick={() => loadDirectory(currentPath)} title="Refresh">
            🔄
          </button>
          {onClose && (
            <button onClick={onClose} title="Close">
              ✕
            </button>
          )}
        </div>
      </div>

      <div className="current-path">
        <span>{currentPath}</span>
        {currentPath !== '/' && (
          <button 
            onClick={() => setCurrentPath(path.dirname(currentPath))}
            title="Go up"
          >
            ⬆️
          </button>
        )}
      </div>

      <div className="file-list">
        {files.map(file => (
          <div 
            key={file.path}
            className={`file-item ${file.is_directory ? 'directory' : 'file'}`}
            onClick={() => handleFileClick(file)}
          >
            <span className="file-icon">
              {file.is_directory ? '📁' : '📄'}
            </span>
            <span className="file-name">{file.name}</span>
            {!file.is_directory && (
              <span className="file-size">
                {formatFileSize(file.size)}
              </span>
            )}
          </div>
        ))}
      </div>
    </div>
  );
};

// Register the panel
export function registerFileExplorerPanel(context: LokusPluginAPI) {
  context.ui.registerPanel('file-explorer', {
    title: 'Files',
    icon: 'Folder',
    position: 'sidebar-left',
    component: FileExplorerPanel,
    defaultWidth: 250,
    resizable: true,
    collapsible: true
  });
}
```

## Advanced Plugin Development

### Error Handling and Debugging

The new plugin architecture includes comprehensive error handling:

```typescript
// src/utils/ErrorHandler.ts
import { PluginErrorHandler } from '@lokus/plugin-api';

export class CustomErrorHandler implements PluginErrorHandler {
  handleError(error: Error, context: string): void {
    // Log error with context
    console.error(`Plugin Error in ${context}:`, error);
    
    // Report to error tracking service
    this.reportError(error, context);
    
    // Show user-friendly notification
    lokus.notifications.error(`Plugin error: ${error.message}`);
  }

  async reportError(error: Error, context: string): Promise<void> {
    // Send to analytics or error tracking
    await fetch('/api/errors', {
      method: 'POST',
      body: JSON.stringify({
        error: error.message,
        stack: error.stack,
        context,
        timestamp: new Date().toISOString()
      })
    });
  }
}

// Use in plugin
export default {
  activate(context: LokusPluginAPI) {
    const errorHandler = new CustomErrorHandler();
    context.errors.setHandler(errorHandler);
  }
};
```

### Plugin Communication

Enhanced event system for inter-plugin communication:

```typescript
// Enhanced event system with namespacing
class PluginEventManager {
  private namespace: string;

  constructor(pluginId: string) {
    this.namespace = `plugin.${pluginId}`;
  }

  emit(event: string, data: any): void {
    lokus.events.emit(`${this.namespace}.${event}`, data);
  }

  on(event: string, callback: (data: any) => void): () => void {
    return lokus.events.on(`${this.namespace}.${event}`, callback);
  }

  onAny(callback: (event: string, data: any) => void): () => void {
    return lokus.events.on(`${this.namespace}.*`, callback);
  }

  // Cross-plugin communication
  subscribe(pluginId: string, event: string, callback: (data: any) => void): () => void {
    return lokus.events.on(`plugin.${pluginId}.${event}`, callback);
  }
}

// Usage in plugin
const events = new PluginEventManager('task-management');

// Emit events
events.emit('task-created', { id: '123', title: 'New Task' });
events.emit('task-updated', { id: '123', status: 'completed' });

// Listen to own events
events.on('task-created', (data) => {
  console.log('New task created:', data);
});

// Listen to other plugins
events.subscribe('calendar', 'event-created', (data) => {
  // Create task from calendar event
  createTaskFromEvent(data);
});
```

import { LokusPlugin, LokusPluginAPI, DataProvider } from '@lokus/plugin-api';

class CalendarDataProvider implements DataProvider {
  id = 'calendar';
  name = 'Calendar Events';

  async authenticate(): Promise<boolean> {
    // Google Calendar OAuth
    return await this.googleAuth();
  }

  async search(query: string): Promise<CalendarEvent[]> {
    const events = await this.fetchEvents();
    return events.filter(event => 
      event.title.toLowerCase().includes(query.toLowerCase())
    );
  }

  async getUpcomingEvents(days: number = 7): Promise<CalendarEvent[]> {
    const start = new Date();

    return await this.fetchEventsInRange(start, end);
  }
